CREATE TABLE transaction (
	trans_id INT NOT NULL -- an ID generated by the visualizer app
	,amount NUMERIC(8, 2) -- the amount of the transaction, positive for additions, negative for subtractions
	,trans_desc VARCHAR(100) -- the description of the transaction as given in the statement, before being modified
	,account_id INT -- the account for which this transaction was made
	,addsub TINYINT -- instead of using debit/credit to complicate things, using addition (1) to classify deposits and subtraction (2) to classify withdrawals
	,trans_type_id INT NOT NULL -- will reference the types table to figure out what this type of transaction is
	,PRIMARY KEY (trans_id)
	);
